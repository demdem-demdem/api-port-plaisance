const Reservation = require("../models/reservations");
const Catway = require("../models/catways");

/**
 * Crée une nouvelle réservation pour un catway spécifique.
 * 
 * @async
 * @function createReservation
 * @param {string} catwayId - L'ID technique du catway concerné.
 * @param {Object} reservationData - Données de la réservation (clientName, boatName, checkIn, checkOut).
 * @param {string} reservationData.clientName - Nom du client.
 * @param {string} reservationData.boatName - Nom du bateau.
 * @param {date} reservationData.checkIn - Heure d'arrivée.
 * @param {data} reservationData.checkOut - Heure de départ.
 * @returns {Promise<Object>} La réservation créée.
 * @throws {Error} Si le catway n'existe pas.
 */
exports.createReservation = async (catwayId, reservationData) => {
  const catway = await Catway.findById(catwayId);
  if (!catway) {
    throw new Error("Catway non trouvé");
  }
  reservationData.catwayNumber = catway.catwayNumber; 
  return await Reservation.create(reservationData);
};

/**
 * Récupère l'ensemble des réservations existantes.
 * 
 * @async
 * @function getAllReservations
 * @returns {Promise<Array<Object>>} Liste de toutes les réservations.
 */
exports.getAllReservations = async () => {
  return await Reservation.find();
};

/**
 * Récupère toutes les réservations associées à un catway précis.
 * 
 * @async
 * @function getAllReservationsForCatway
 * @param {string} catwayId - L'ID technique du catway.
 * @returns {Promise<Array<Object>>} Liste des réservations pour ce catway.
 * @throws {Error} Si le catway n'est pas trouvé.
 */
exports.getAllReservationsForCatway = async (catwayId) => {
  const catway = await Catway.findById(catwayId);
  if (!catway) {
    throw new Error("Catway non trouvé");
  }
  return await Reservation.find({ catwayNumber: catway.catwayNumber });
};

/**
 * Récupère une réservation spécifique liée à un catway spécifique.
 * 
 * @async
 * @function getReservationByIdForCatway
 * @param {string} catwayId - L'ID technique du catway.
 * @param {string} reservationId - L'ID technique de la réservation.
 * @returns {Promise<Object>} La réservation trouvée.
 * @throws {Error} Si le catway ou la réservation n'existe pas.
 */
exports.getReservationByIdForCatway = async (catwayId, reservationId) => {
  
  const catway = await Catway.findById(catwayId );
  if (!catway) {
    throw new Error("Catway non trouvé");
  }
  const reservation = await Reservation.findOne({
    _id: reservationId,
    catwayNumber: catway.catwayNumber,
  });
  if (!reservation) {
    throw new Error("Réservation non trouvée pour ce catway");
  }
  return reservation;
};

/**
 * Supprime une réservation spécifique d'un catway.
 * 
 * @async
 * @function deleteReservationForCatway
 * @param {string} catwayId - L'ID technique du catway.
 * @param {string} reservationId - L'ID technique de la réservation à supprimer.
 * @returns {Promise<{message: string}>} Confirmation de suppression.
 * @throws {Error} Si le catway ou la réservation n'est pas trouvé.
 */
exports.deleteReservationForCatway = async (catwayId, reservationId) => {
  const catway = await Catway.findById(catwayId);
  if (!catway) {
    throw new Error("Catway non trouvé");
  }
  const result = await Reservation.deleteOne({
    _id: reservationId,
    catwayNumber: catway.catwayNumber,
  });
  if (result.deletedCount === 0) {
    throw new Error("Réservation non trouvée pour ce catway");
  }
  return { message: "Reservation deleted successfully" };
};
